const gulp = require('gulp');
const sass = require('gulp-sass');
const autoprefixer = require('gulp-autoprefixer');
const cssnano = require('gulp-cssnano');

// SASS compilation
gulp.task('sass', () => {
  return gulp.src('src/**/*.scss')
    .pipe(sass().on('error', sass.logError))
    .pipe(autoprefixer())
    .pipe(gulp.dest('css/'));
});

// Watch for changes
gulp.task('watch', () => {
  gulp.watch('src/**/*.scss', gulp.series('sass'));
  gulp.watch('src/**/*.html', gulp.series('html'));
});

ğŸ¯ Complete Route Creation Process

Step 1: Create Controller Action

Create or modify a controller in src/Core/Controllers/ directory.

Example: HomeController.php 
<?php
namespace App\Core\Controllers;

use App\Core\Http\Controller;

class HomeController extends Controller
{
    public function index()
    {
        // Data passed here will be available in template as $this->parameters
        $this->parameters = [
            'title' => 'Welcome to My App',
            'description' => 'A beautiful landing page',
            'posts' => [
                ['title' => 'First Post', 'content' => 'Content here'],
                ['title' => 'Second Post', 'content' => 'More content']
            ]
        ];
        
        // Return view name (corresponds to template file)
        return 'home/index';
    }
    
    public function about()
    {
        $this->parameters = [
            'title' => 'About Us',
            'team' => ['John', 'Jane', 'Bob']
        ];
        
        return 'home/about';
    }
}

Step 2: Create Template File (.tpl)

Create a .tpl file in src/Core/templates/{controller}/{action}.tpl

Example: src/Core/templates/home/index.tpl 

                  #app | prepend = <?php $data = $this->parameters;?>
              
              h1[data-v-title] | innerText = <?php echo $data['title']; ?>
              h3[data-v-description] | innerText = <?php echo $data['description']; ?>
              p[data-v-intro] | innerText = <?php echo $data['intro']; ?>
              
              #instructions | prepend = <?php foreach($data['instructions'] as $instruction) { ?>
                  li[data-v-instruction] | innerHTML = <?php echo $instruction; ?>
              #instructions | append = <?php }?>
              
              a[data-v-action] | innerText = <?php echo $data['button_text']; ?>

Step 3: Create HTML Source File

Create the HTML file in themes/landing/src/{controller}/{action}.html

Example: themes/landing/src/home/index.html 
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Home Page</title>
</head>
<body>
    <div id="app">
        <!-- This is where your .tpl content will be injected -->
        <!-- The #app selector tells the system where to append content -->
    </div>
</body>
</html>

Step 4: Auto-Generation Process

The system automatically:

  1. Reads from themes/landing/src/ folder
  2. Processes the .tpl template
  3. Generates final HTML in themes/landing/{controller}/ folder
  4. Injects processed content into the HTML file

Template Syntax

  • #app | append = - Tells the system to append content to #app element
  • $this->parameters - Contains all data passed from controller
  • Heredoc syntax - Use <<<HTML for clean HTML writing
  • Variable interpolation - Use {$variable} inside heredoc

âš ï¸ Important Notes
